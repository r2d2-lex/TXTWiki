git init --bare - создание чистого репозитория
git add . - добавление всех файлов из текущего каталога в репу
git commit -am "Text" - добавление коммита
git push - добавить изменения в репозиторий

git pull - скачать изменения сделаные другими пользователями
git branch ИмяВетки 4e91f95ab53645aa51b7e19b0ddf3b85e6b09bee - создать новую ветку с ID
git checkout master - переключиться на ветку master


origin  - удалённый репозиторий
git merge - слияние веток коммитов

# Вернуть файл file.py к последнему коммиту
git checkout -- file.py

Мердж ветки в мастер
Выполняется после завершения работы над своей веткой при помощи команды git merge. Чтобы вмерджить ветку в мастер, нужно сначала перейти в мастер, а затем выполнить git merge branch_name.
$ git checkout master
$ git merge news
